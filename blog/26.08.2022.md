```py
from turtle import *

def mini():
    forward(5)
    left(60)
    forward(5)
    right(120)
    forward(5)
    left(60)
    
def normal():
    forward(5)
    left(60)
    mini()
    forward(5)
    right(120)
    mini()
    forward(5)
    left(60)

def normal_ohni_forward():
    left(60)
    mini()
    forward(5)
    right(120)
    mini()
    forward(5)
    left(60)

def maxi():
    mini()
    normal()
    mini()
    forward(5)
    left(60)
    mini()
    normal()
    right(60)
    normal_ohni_forward()
    right(60)
    normal_ohni_forward()
    mini()
    forward(5)
    left(60)
    mini()
    normal()
    mini()
    right(60)


penup()
goto(0, 100)
pendown()
for counter in range(6):
    maxi()
    
```
